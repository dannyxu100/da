 14 		inPageArea: function(pos,size){
 15 				if( null == da.isNull( pos, null ) ) return "5";
 16 				
 17 				var da_Win = da(window),
 18 						winWidth = parseInt( da_Win.width(), 10 ) || 0,
 19 						winHeight = parseInt( da_Win.height(), 10 ) || 0;
 20 				
 21 				var d_toLeft = pos.left,
 22 						d_toRight = winWidth - pos.left,
 23 						d_toTop = pos.top,
 24 						d_toBottom = winHeight - pos.top;
 25 				
 26 				var sArea = [];
 27 		  	if( d_toTop>=d_toBottom )	sArea.push( "B" );				//偏下
 28 		  	else	sArea.push( "T" );														//偏上
 29 			  if( d_toRight>=d_toLeft )	sArea.push( "L" );				//偏左
 30 			  else	sArea.push( "R" );														//偏右
 31 		  	
 32 		  	if( undefined == size ) return sArea.join("");
 33 
 34 		  	if( size.height >= size.width )			//如果信息框为 长条形
 35 		  		sArea.reverse();
 36 		  	
 37 		  	var nArea = 5;				//默认正中
 38 		  	switch( sArea.join("") ){
 39 		  		case "TL": nArea = "87"; break;
 40 		  		case "LT": nArea = "47"; break;
 41 		  		case "TR": nArea = "89"; break;
 42 		  		case "RT": nArea = "69"; break;
 43 		  		case "BL": nArea = "21"; break;
 44 		  		case "LB": nArea = "41"; break;
 45 		  		case "BR": nArea = "23"; break;
 46 		  		case "RB": nArea = "63"; break;
 47 		  	}
 48 		  	
 49 		  	return nArea;
